<script>

/* 🔥 SEGURANÇA */
let logado = localStorage.getItem("logado");
let tipo = localStorage.getItem("tipo");

if(!logado || tipo !== "admin"){
  window.location.href = "login.html";
}

/* USER */
let usuario = localStorage.getItem("usuario") || "ADMIN";

userBox.innerHTML = `
<strong>${usuario}</strong>
<small>ADMIN</small>
`;

/* 🔥 ABRIR PÁGINA */
function abrir(pagina, el){

  document.getElementById("frame").src = pagina;

  // salva última tela
  localStorage.setItem("pagina_admin", pagina);

  document.querySelectorAll(".menu-item")
    .forEach(i=>i.classList.remove("active"));

  el.classList.add("active");
}

/* SUBMENU */
function toggleMenu(el){
  el.nextElementSibling.classList.toggle("open");
}

/* 🔥 INIT (CORREÇÃO DO BUG BRANCO) */
window.onload = function(){

  let ultima = localStorage.getItem("pagina_admin");

  // se não tiver, abre dashboard
  let pagina = ultima || "admin-dashboard.html";

  document.getElementById("frame").src = pagina;
}

/* LOGOUT */
function logout(){
  localStorage.clear();
  window.location.href = "login.html";
}

</script>
